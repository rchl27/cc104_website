1. Linear Search

Start
Input array A and target T
For i = 0 to length(A)-1
    If A[i] == T
        Output i
        Stop
EndFor
Output "Not Found"
End

2. Binary Search

Start
Set low = 0, high = n-1
While low ≤ high
    mid = (low+high)/2
    If A[mid] == T
        Output mid
        Stop
    Else if A[mid] < T
        low = mid + 1
    Else
        high = mid - 1
EndWhile
Output "Not Found"
End

3. Bubble Sort

Start
Repeat
    swapped = false
    For i = 0 to n-2
        If A[i] > A[i+1]
            Swap
            swapped = true
    EndFor
Until swapped = false
End

4. Selection Sort

Start
For i = 0 to n-2
    min = i
    For j = i+1 to n-1
        If A[j] < A[min]
            min = j
    EndFor
    Swap A[i], A[min]
EndFor
End

5. Insertion Sort

Start
For i = 1 to n-1
    key = A[i]
    j = i - 1
    While j ≥ 0 AND A[j] > key
        A[j+1] = A[j]
        j = j - 1
    EndWhile
    A[j+1] = key
EndFor
End

6. Fibonacci

Function Fibo(n)
    If n ≤ 1 return n
    Return Fibo(n-1) + Fibo(n-2)
EndFunction

7. Factorial

Function Fact(n)
    If n == 0 return 1
    Return n * Fact(n-1)
EndFunction

8. Find Maximum

max = A[0]
For i = 1 to n-1
    If A[i] > max
        max = A[i]
EndFor
Output max

9. Find Minimum

min = A[0]
For i = 1 to n-1
    If A[i] < min
        min = A[i]
EndFor
Output min

10. Sum of Array

sum = 0
For i = 0 to n-1
    sum = sum + A[i]
EndFor
Output sum
